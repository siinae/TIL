# 기본자료형, 자료형과 연산, 문자열



1. 기본 자료형

```python
#변수명 = 변수값
name = 'sinae' 
age = 20
height = 160.0
ischild = age < 20

#변수값,자료형 호출
print(age)
#20

print(type(age))
#<class 'int'>

print(type(height))
#<class 'float'>

print(type(ischild))
#<class 'bool'>

#쌍따옴표, 홑따옴표 포함 문자열 사용
a = "'낮말'은 새가 듣고 '밤말'은 쥐가 듣는다."
print(a)
#'낮말'은 새가 듣고 '밤말'은 쥐가 듣는다.

b = '"시간은 금"이라는 말이 있다.'
print(b)
#"시간은 금"이라는 말이 있다.

#자료형 변환
age = int('20')  #int(value): int형으로 변환
type(age)
#int

age = str(20)    #str(value): str형으로 변환
type(age)
#str


age = float(20)  #float(value): float형으로 변환
print(age, type(age))
#20.0 <class 'float'>

#input으로 입력받은 값을 숫자형으로 변환
a = int(input('나이:'))
#나이 = 20

type(a)
#int
```



2. 자료형과 연산

```python
#산술연산
a,b = 3,4

print('a+b=',a+b)
#a+b= 7
print('a-b=',a-b)
#a-b= -1
print('a*b=',a*b)
#a*b= 12
print('a/b=',a/b)
#a/b= 0.75
print('a//b=',a//b)
#a//b= 0
print('a%b=',a%b)
#a%b= 3
print('a**b=',a**b)
#a**b= 81

#문자열 연산
a = 'good'
b = 'morning'
a+b
#'goodmorning'

s = 'ha'
s * 3
#'hahaha'

english = 80
result = '영어점수' + str(english)
print(result)
#영어점수80

#할당연산자
a = 1

a = b = c = 1 #a,b,c에 모두 1을 할당
print(a,b,c) 
#1 1 1

a,b,c = 1,2,3 #a,b,c에 각각 1,2,3 할당
print(a,b,c)
#1 2 3

#복합할당연산자
x = 10
x += 20 #x = x + 20 = 10 + 20 = 30
print(x)
#30

x = 3
y = 5
x *= x+y  #x = x * (x+y) = 3 * (3 + 5) = 24
print(x)
#24

#f스트링으로 출력하기
name = input('이름:')
print(name,'님, 안녕하세요?')
#이름: sinae
#sinae 님, 안녕하세요?

print(f'{name}님 안녕하세요') #명령어 f를 스트링의 따옴표 시작 전에 넣어 중간중간에 변수의 값을 출력하도록 함
#sinae님 안녕하세요

age = int(input('나이:'))
print('나이는',age,'살이시군요. 내년이면',age+1,'살이 되시겠네요')
#나이: 20
#나이는 20 살이시군요. 내년이면 21 살이 되시겠네요

print(f'나이는 {age}살이시군요. 내년이면 {age+1}살이 되시겠네요')
#나이는 20살이시군요. 내년이면 21살이 되시겠네요

#두 수를 입력받아 더하는 프로그램 작성
a = int(input('첫번째숫자:'))
b = int(input('두번째숫자:'))

print(a,'+',b,'=',a+b)
#첫번째숫자: 1
#두번째숫자: 3
1 + 3 = 4

#섭씨 온도 변환 프로그램 작성

#1. 화씨온도 입력
f = float(input('화씨온도:'))

#2. 섭씨온도로 계산하기
c = (f - 32) * (5/9)

#3. 결과 출력
print(f'화씨온도:{f} --> 섭씨온도:{c}')

#화씨온도:100
#화씨온도:100.0 --> 섭씨온도:37.77777777777778
        
```



3. 문자열 

```python
#문자열 인덱싱

#양수 인덱스 사용하기

a = 'Hello World'

#'H' 출력하기
print(a[0])

#'W' 출력하기
print(a[6])

#'d' 출력하기
print(a[10])

#음수 인덱스 사용하기

#'H' 출력하기
print(a[-11])

#'W' 출력하기
print(a[-5])

#'d' 출력하기
print(a[-1])

#범위 밖의 인덱스 사용시 IndexError

#문자열 슬라이싱

a = 'Hello World'

#양수

# 'Hello' 출력하기
print(a[0:5])

# 'World' 출력하기
print(a[6:11])

#음수

# 'Hello' 출력하기
print(a[-11:-6])

# 'World' 출력하기
print(a[-5:])

#역순 출력
a = 'Hello World'
print(a[::-1])
#간격을 마이너스(-)로 하면 역순 출력
#dlroW olleH

#문자열 교체 함수
#문자열.replace(찾을문자열, 교체할 문자열)

a = '나는 양배추 좋아. 양배추 최고!'

#양배추 -> 떡볶이로 교체
print(a.replace('양배추','떡볶이'))
#나는 떡볶이 좋아. 떡볶이 최고!

#문자 위치 찾기 함수
#문자열.find(문자)

a = 'Hello, Python!'

#'e'의 위치 찾기
print(a.find('e'))
#1

#'l'의 위치 찾기
print(a.find('l'))
#2

#소문자로 변환하기
#문자열.lower()

a = 'Hello, Python!!!'
a.lower()
#'hello, python!!!'

#대문자로 변환하기
#문자열.upper()

a = 'Hello, Python!!!'
a.upper()
#'HELLO, PYTHON!!!'

#문자열 나누기
#문자열.split(구분자)

phone = '010-123-4567'
phone.split('-')
#['010', '123', '4567']

#사용자의 영문 이름을 입력 받아 성과 이름 순서를 바꾸어 출력하는 프로그램 작성
#(성과 이름은 공백으로 구분)

#1. 사용자의 영문이름 입력받기
full_name = input('영문이름(성과 이름을 공백으로 구분해주세요):')

#2. 공백의 위치 찾기
space = full_name.find('')
print(space)

#3. 성, 이름을 슬라이싱하여 각각 변수에 담기
first_name = full_name[:space]
last_name = full_name[space+1:]

print(first_name)
print(last_name)

#4. 성, 이름을 바꾸어 출력
print(last_name, first_name)

#영문이름(성과 이름을 공백으로 구분해주세요): sinae Lee
#5
#sinae
#Lee
#Lee sinae

```
